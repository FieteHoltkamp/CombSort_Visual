<body>

    <h1>CombSort Visualisierung</h1>

    <div class="controls">
        <button id="startBtn" onclick="startSorting()">Sortieren starten</button>
        <button id="resetBtn" onclick="resetArray()">Neu generieren</button>
        <div>
            <label for="speed">Verzögerung: <span id="speedVal">50</span> ms</label>
            <input type="range" id="speed" min="5" max="300" value="50" oninput="updateSpeed(this.value)">
        </div>
    </div>

    <div class="container" id="arrayContainer"></div>

    <script>
        const container = document.getElementById('arrayContainer');
        const startBtn = document.getElementById('startBtn');
        const resetBtn = document.getElementById('resetBtn');
        const speedVal = document.getElementById('speedVal');
        
        const ARRAY_SIZE = 50;
        let array = [];
        let delay = 50;
        let isSorting = false;

        // Array beim Start generieren
        resetArray();

        function resetArray() {
            if (isSorting) return;
            container.innerHTML = '';
            array = [];
            for (let i = 0; i < ARRAY_SIZE; i++) {
                // Zufällige Werte zwischen 10 und 380 (für die Pixelhöhe passend)
                array.push(Math.floor(Math.random() * 370) + 10);
            }
            createBars();
        }

        function createBars() {
            container.innerHTML = '';
            for (let i = 0; i < ARRAY_SIZE; i++) {
                const bar = document.createElement('div');
                bar.classList.add('bar');
                bar.style.height = `${array[i]}px`;
                bar.id = `bar-${i}`;
                container.appendChild(bar);
            }
        }

        function updateSpeed(val) {
            delay = val;
            speedVal.innerText = val;
        }

        // Hilfsfunktion für die Animation/Verzögerung
        function sleep(ms) {
            return new Promise(resolve => setTimeout(resolve, ms));
        }

        async function startSorting() {
            if (isSorting) return;
            isSorting = true;
            startBtn.disabled = true;
            resetBtn.disabled = true;

            await combSort();

            isSorting = false;
            startBtn.disabled = false;
            resetBtn.disabled = false;
        }

        // Der CombSort Algorithmus
        async function combSort() {
            let gap = array.length;
            const shrink = 1.3;
            let sorted = false;

            while (!sorted) {
                // Update den Abstand (Gap)
                gap = Math.floor(gap / shrink);
                if (gap <= 1) {
                    gap = 1;
                    sorted = true; // Annahme, dass es die letzte Runde sein könnte
                }

                for (let i = 0; i + gap < array.length; i++) {
                    const idx1 = i;
                    const idx2 = i + gap;

                    const bar1 = document.getElementById(`bar-${idx1}`);
                    const bar2 = document.getElementById(`bar-${idx2}`);

                    // Visuell: Rot einfärben (Vergleich)
                    bar1.classList.add('comparing');
                    bar2.classList.add('comparing');

                    await sleep(delay);

                    if (array[idx1] > array[idx2]) {
                        // Werte im Array tauschen
                        let temp = array[idx1];
                        array[idx1] = array[idx2];
                        array[idx2] = temp;

                        // Höhen der Balken tauschen
                        bar1.style.height = `${array[idx1]}px`;
                        bar2.style.height = `${array[idx2]}px`;

                        // Da getauscht wurde, ist es noch nicht garantiert fertig sortiert
                        sorted = false; 
                    }

                    // Visuell: Rot wieder entfernen
                    bar1.classList.remove('comparing');
                    bar2.classList.remove('comparing');
                }

                // Hinweis zu CombSort: Sobald Gap == 1 ist, verhält es sich wie BubbleSort.
                // Wenn ein Durchlauf mit Gap == 1 ohne Tausch (sorted == true) durchläuft, ist alles fertig.
            }

            // Am Ende alle Balken grün einfärben (erfolgreich sortiert)
            for (let i = 0; i < array.length; i++) {
                document.getElementById(`bar-${i}`).classList.add('sorted');
                await sleep(10); // Schöner "Sweep"-Effekt am Ende
            }
        }
    </script>
